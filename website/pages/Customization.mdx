3. Use the [`dock`](/) function in your **client** components:

```tsx {10-16}
"use client";

import { dock } from "@watercubz/dockly";

import { DockOption, DockItems, icons } from "./utils/options";

export default function MyComponent() {
  return (
    <div>
      <Dock
        DockItems={DockItems}
        DockOption={DockOption}
        icon1={icons[0]}
        icon2={icons[1]}
        icon3={icons[2]}
      />
    </div>
  );
}
```
